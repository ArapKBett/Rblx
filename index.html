-- Advanced weapon system example
local WeaponSystem = {}

function WeaponSystem:Fire(player, target)
    local damage = self:CalculateDamage(player)
    local hitSuccess = self:CheckHit(player, target)
    
    if hitSuccess then
        self:ApplyDamage(target, damage)
        self:CreateVFX(player, target)
        self:PlaySFX("Hit")
    end
end

return WeaponSystem
